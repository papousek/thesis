	\begin{itemize}
		\item rating of chess players
		\item player 1 vs. player 2\\ $\rightarrow$ student ($\theta$) vs. item ($d$)
		\item update, $R \in \{0, 1\}$, $E(R) \in [0, 1]$
		 \begin{itemize}
			\item $ \theta := \theta + K \cdot (R - E(R))$
			\item $ d := d - K \cdot (R - E(R))$
		 \end{itemize}
	\end{itemize}
